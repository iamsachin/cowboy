    ```typescript
    import { ref, type Ref, onMounted, onUnmounted } from 'vue';

    export function useScrollTracker(containerRef: Ref<HTMLElement | null>) {
      const isAtBottom = ref(true);
      const THRESHOLD = 100;
      let rafId: number | null = null;

      function checkBottom(): void { ... }
      function onScroll(): void { ... } // RAF-throttled passive scroll listener
      function scrollToBottom(smooth?: boolean): void { ... }

      // Call BEFORE data update, returns a restore function to call after nextTick
      function captureScrollPosition(): (() => void) | null { ... }

      onMounted(() => { containerRef.value?.addEventListener('scroll', onScroll, { passive: true }); });
      onUnmounted(() => { containerRef.value?.removeEventListener('scroll', onScroll); if (rafId) cancelAnimationFrame(rafId); });

      return { isAtBottom, scrollToBottom, checkBottom, captureScrollPosition };
    }
    ```

    Key implementation details:
    - `checkBottom()`: `isAtBottom.value = el.scrollHeight - el.scrollTop - el.clientHeight <= THRESHOLD`
    - `onScroll()`: RAF-throttled — if `rafId` is set, skip. Otherwise `rafId = requestAnimationFrame(() => { rafId = null; checkBottom(); })`
    - `scrollToBottom(smooth = false)`: `el.scrollTo({ top: el.scrollHeight, behavior: smooth ? 'smooth' : 'instant' })`
    - `captureScrollPosition()`: Captures `prevScrollHeight` and `prevScrollTop`. Returns a restore function that computes `delta = el.scrollHeight - prevScrollHeight` and sets `el.scrollTop = prevScrollTop + delta`. Only restores if user was NOT at bottom (if at bottom, auto-scroll via `scrollToBottom()` instead).

    - Test that captureScrollPosition auto-scrolls when at bottom instead of preserving

    Note: happy-dom has limited scroll API support. Mock the element properties directly rather than relying on happy-dom's scroll simulation.
