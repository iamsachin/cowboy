## NEW contracts this plan introduces

### Backend — GET /api/analytics/subagents/recent
Query params: `limit` (u32, default 100, max 200).
Response JSON (camelCase):
```json
{
  "rows": [
    {
      "conversationId": "<parent_conversation_id>",
      "toolCallId": "<tool_call_id>",
      "description": "Fix login bug",
      "status": "success" | "error" | "interrupted",
      "timestamp": "2026-04-17T12:34:56Z"
    }
  ],
  "total": 42
}
```
Rules:
- SELECT FROM tool_calls WHERE `subagent_summary IS NOT NULL` AND (name='Task' OR name='Agent').
- ORDER BY created_at DESC.
- `description`: parse `input` JSON on the Rust side; fall back to
  `input.prompt` (truncated to 80 chars) then `'Subagent'`.
- `status`: parse `subagent_summary` JSON, take `.status` field.
- `timestamp`: `created_at` column verbatim.
- Limit clamped to [1, 200].

### Route semantics
- Selecting a `current-subagent` entry: calls an injected callback (exposed by the
  page) `onCurrentSubagentSelect?: (toolCallId: string) => void`. No router push.
- Selecting a `recent-subagent` entry: `router.push({
    path: `/conversations/${item.conversationId}`,
    query: { jump: item.toolCallId }
  })` then close().
- ConversationDetailPage watches `route.query.jump`; once `data.value` is populated,
  it calls `handleSubagentJump(jump)` and clears the query with
  `router.replace({ query: {} })` to prevent re-trigger on re-mount.

### Query syntax
- `sub` / `sub N` (numeric `N`): within the current-conversation section,
  match by 1-based index. `sub` alone with no number = show all.
- Non-numeric query: matches both descriptions in current-conversation list AND
  recent-subagent list (case-insensitive substring match, no Fuse for simplicity —
  Fuse was overkill for short keyword hits here and adds no value).
- Empty query: both sections render their full lists (up to 10 each).

    use axum::{extract::{Query, State}, routing::get, Json, Router};
    use serde::{Deserialize, Serialize};
    use tokio_rusqlite::rusqlite::params;

    use crate::error::AppError;
    use crate::server::AppState;

    pub fn routes() -> Router<AppState> {
        Router::new()
            .route("/api/analytics/subagents/recent", get(recent))
    }

    #[derive(Deserialize)]
    pub struct RecentParams {
        pub limit: Option<u32>,
    }

    #[derive(Serialize)]
    #[serde(rename_all = "camelCase")]
    struct RecentSubagentRow {
        conversation_id: String,
        tool_call_id: String,
        description: String,
        status: String,         // "success" | "error" | "interrupted"
        timestamp: String,      // ISO8601, matches tool_calls.created_at verbatim
    }

    async fn recent(
        State(state): State<AppState>,
        Query(params): Query<RecentParams>,
    ) -> Result<Json<serde_json::Value>, AppError> {
        let requested = params.limit.unwrap_or(100);
        let limit: i64 = requested.clamp(1, 200) as i64;

        let rows: Vec<RecentSubagentRow> = state
            .db
            .call(move |conn| {
                let mut stmt = conn.prepare(
                    "SELECT tc.conversation_id, tc.id, tc.input, tc.subagent_summary, tc.created_at
                     FROM tool_calls tc
                     WHERE tc.subagent_summary IS NOT NULL
                       AND (tc.name = 'Task' OR tc.name = 'Agent')
                     ORDER BY tc.created_at DESC
                     LIMIT ?1"
                )?;
                let iter = stmt.query_map(params![limit], |row| {
                    let conv_id: String = row.get(0)?;
                    let tc_id: String = row.get(1)?;
                    let input_str: Option<String> = row.get(2)?;
                    let summary_str: Option<String> = row.get(3)?;
                    let created_at: String = row.get(4)?;
                    Ok((conv_id, tc_id, input_str, summary_str, created_at))
                })?;

                let mut out = Vec::<RecentSubagentRow>::with_capacity(limit as usize);
                for r in iter {
                    let (conversation_id, tool_call_id, input_str, summary_str, timestamp) = r?;

                    // Derive description: input.description (string) ->
                    // input.prompt (trunc 80) -> "Subagent" fallback.
                    let description = input_str
                        .as_deref()
                        .and_then(|s| serde_json::from_str::<serde_json::Value>(s).ok())
                        .map(|v| {
                            if let Some(d) = v.get("description").and_then(|x| x.as_str()) {
                                return d.to_string();
                            }
                            if let Some(p) = v.get("prompt").and_then(|x| x.as_str()) {
                                let truncated: String = p.chars().take(80).collect();
                                return if p.chars().count() > 80 {
                                    format!("{}...", truncated)
                                } else {
                                    truncated
                                };
                            }
                            "Subagent".to_string()
                        })
                        .unwrap_or_else(|| "Subagent".to_string());

                    // Derive status: summary.status (any of success/error/interrupted).
                    // Fallback to "success" if parse fails (should never happen — WHERE
                    // clause requires non-null summary — but avoid panics).
                    let status = summary_str
                        .as_deref()
                        .and_then(|s| serde_json::from_str::<serde_json::Value>(s).ok())
                        .and_then(|v| v.get("status").and_then(|x| x.as_str()).map(String::from))
                        .unwrap_or_else(|| "success".to_string());

                    out.push(RecentSubagentRow {
                        conversation_id,
                        tool_call_id,
                        description,
                        status,
                        timestamp,
                    });
                }
                Ok(out)
            })
            .await?;

        let total = rows.len() as u32;
        Ok(Json(serde_json::json!({ "rows": rows, "total": total })))
    }
    ```

       interface ParsedQuery {
         mode: 'normal' | 'sub-index' | 'sub-keyword';
         subIndex?: number;   // 1-based
         keyword?: string;
       }
       function parseQuery(q: string): ParsedQuery {
         const trimmed = q.trim();
         if (trimmed.length === 0) return { mode: 'normal' };
         const m = /^sub(?:agent)?\s*#?(\d+)?$/i.exec(trimmed);
         if (m) {
           return m[1]
             ? { mode: 'sub-index', subIndex: Number(m[1]) }
             : { mode: 'sub-keyword', keyword: '' };
         }
         // Plain keyword — applies to both current AND recent sub-agent lists
         return { mode: 'sub-keyword', keyword: trimmed.toLowerCase() };
       }
       ```

       const filteredCurrentSubagents = computed<{ index: number; entry: SubagentListEntry }[]>(() => {
         const list = currentConversationSubagents.value;
         if (list.length === 0) return [];
         const parsed = parseQuery(query.value);
         const indexed = list.map((entry, i) => ({ index: i + 1, entry }));
