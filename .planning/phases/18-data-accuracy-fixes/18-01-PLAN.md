```typescript
const messageDuration = computed(() => {
  if (!data.value || !data.value.messages || data.value.messages.length < 2) return '--';
  const sorted = [...data.value.messages].sort(
    (a, b) => new Date(a.createdAt).getTime() - new Date(b.createdAt).getTime()
  );
  const first = new Date(sorted[0].createdAt).getTime();
  const last = new Date(sorted[sorted.length - 1].createdAt).getTime();
  return formatDuration(sorted[0].createdAt, sorted[sorted.length - 1].createdAt);
});
```
