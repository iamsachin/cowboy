```typescript
// Events: { type: string, seq: number, ...payload }
// On connect: sends { type: 'connected' }
// broadcastEvent adds seq to payload and sends to all clients
let seq = 0;
app.decorate('broadcastEvent', (event) => {
    const payload = JSON.stringify({ ...event, seq: ++seq });
    for (const client of app.websocketServer.clients) {
        if (client.readyState === WebSocket.OPEN) client.send(payload);
    }
});
```
