// Module-level debounce timer
let wsDebounceTimer: ReturnType<typeof setTimeout> | null = null;
function debouncedWsRefetch(): void {
  if (wsDebounceTimer) clearTimeout(wsDebounceTimer);
  wsDebounceTimer = setTimeout(() => { wsDebounceTimer = null; fetchAll(true); }, 500);
}
const { on } = useWebSocket();
on('conversation:changed', debouncedWsRefetch);
on('conversation:created', debouncedWsRefetch);
on('system:full-refresh', debouncedWsRefetch);

    8. Export `currentInput`, `currentOutput` from the composable as well
    9. Fill missing minutes: generate all 60 minute slots and fill gaps with zeros so the chart shows a complete timeline (per research open question #2). Export as `filledTokenRate` computed ref.
    10. Add module-level `dismissed = ref(localStorage.getItem('token-widget-dismissed') === 'true')` plus `dismiss()` and `restore()` functions that update both the ref and localStorage. Export these from the composable for shared state between LiveTokenPill and AppSidebar.

    4. Chart options: `animation: false` (per established pattern), responsive, maintainAspectRatio false, scales with token count formatting
